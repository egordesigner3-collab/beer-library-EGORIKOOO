  <div class="controls">
    <input type="text" id="search" placeholder="🔍 Поиск по названию или бренду...">
    <select id="f-country"><option value="">🌍 Все страны</option></select>

  <div class="grid" id="grid"></div>

  <script>
    // ==========================================
    // 🍺 ТВОЯ БАЗА ДАННЫХ (РЕДАКТИРУЙ НИЖЕ)
    // ==========================================
    const beers = [
      {
        id: 1,
        name: "Schlossbacher Hefe-Weizen",
        brand: "Schlossbacher",
        country: "Германия",
        style: "Хефевайцен",
        abv: 5.4,
        rating: 5,
        repeat: true,
        // Замени ссылку на свою картинку
        img: "https://images.unsplash.com/photo-1608270586620-2485246391d8?auto=format&fit=crop&w=600&q=80",
        taste: "Вкус свежий, мягкий, с насыщенными пшеничными тонами. Фруктово-дрожжевое послевкусие.",
        where: "О'Кей, специализированные пивные"
      },
      {
        id: 2,
        name: "Guinness Draught",
        brand: "Guinness",
        country: "Ирландия",
        style: "Стаут",
        abv: 4.2,
        rating: 5,
        repeat: true,
        img: "https://images.unsplash.com/photo-1571613316887-6f8d5cbf7ef7?auto=format&fit=crop&w=600&q=80",
        taste: "Классика. Кофейные и шоколадные ноты, кремовая плотная пена.",
        where: "Практически везде"
      },
      {
        id: 3,
        name: "Punk IPA",
        brand: "BrewDog",
        country: "Шотландия",
        style: "IPA",
        abv: 5.4,
        rating: 4,
        repeat: false,
        img: "https://images.unsplash.com/photo-1535958636474-b021ee887813?auto=format&fit=crop&w=600&q=80",
        taste: "Горьковатый, цитрусовый, тропический хмель. Для любителей яркого характера.",
        where: "Красное & Белое, онлайн"
      }
      // 👇 ЧТОБЫ ДОБАВИТЬ НОВОЕ ПИВО:
      // Скопируй блок от { до }, вставь ниже и поменяй данные.
    ];

    // ==========================================
    // ⚙️ ЛОГИКА (НЕ ТРОГАЙ, ЕСЛИ НЕ УВЕРЕН)
    // ==========================================
    const grid = document.getElementById('grid');
    const searchInput = document.getElementById('search');
    const fCountry = document.getElementById('f-country');
    const fStyle = document.getElementById('f-style');
    const fRating = document.getElementById('f-rating');

    // Заполняем фильтры уникальными значениями
    const countries = [...new Set(beers.map(b => b.country))].sort();
    const styles = [...new Set(beers.map(b => b.style))].sort();
    const ratings = [...new Set(beers.map(b => b.rating))].sort((a, b) => b - a);

    countries.forEach(c => fCountry.innerHTML += `<option value="${c}">${c}</option>`);
    styles.forEach(s => fStyle.innerHTML += `<option value="${s}">${s}</option>`);
    ratings.forEach(r => fRating.innerHTML += `<option value="${r}">${r}/5 ⭐</option>`);

    function render() {
      grid.innerHTML = '';
      const q = searchInput.value.toLowerCase().trim();
      const c = fCountry.value;
      const s = fStyle.value;
      const r = fRating.value;

      const filtered = beers.filter(b => {
        const matchQ = !q || b.name.toLowerCase().includes(q) || b.brand.toLowerCase().includes(q);
        const matchC = !c || b.country === c;
        const matchS = !s || b.style === s;
        const matchR = !r || b.rating >= parseInt(r);
        return matchQ && matchC && matchS && matchR;
      });

      if (filtered.length === 0) {
        grid.innerHTML = '<div class="no-results">Ничего не найдено 🍺<br>Попробуй изменить фильтры</div>';
        return;
      }

      filtered.forEach(b => {
        const tried = localStorage.getItem(`beer_tried_${b.id}`) === 'true';
        const card = document.createElement('div');
        card.className = 'card';
        card.innerHTML = `
          <img src="${b.img}" class="card-img" alt="${b.name}" onerror="this.src='https://via.placeholder.com/600x400?text=Нет+фото'">
          <div class="card-body">
            <div class="card-title">${b.name}</div>
            <div class="card-brand">${b.brand} • ${b.abv}%</div>
            <div class="tags">
              <span class="tag rating">⭐ ${b.rating}/5</span>
              <span class="tag">${b.style}</span>
              ${b.repeat ? '<span class="tag repeat"> Повторить</span>' : ''}
            </div>
            <div class="card-desc">
              <b>Где:</b> ${b.where}<br>
              <b>Вкус:</b> ${b.taste}
            </div>
            <button class="btn-tried ${tried ? 'active' : ''}" data-id="${b.id}">
              ${tried ? '✅ Пробовал' : '🍺 Пробовал?'}
            </button>
          </div>
        `;
        grid.appendChild(card);
      });

      // Навешиваем обработчики на кнопки
      document.querySelectorAll('.btn-tried').forEach(btn => {
        btn.addEventListener('click', (e) => {
          const id = e.currentTarget.dataset.id;
          const isTried = localStorage.getItem(`beer_tried_${id}`) === 'true';
          localStorage.setItem(`beer_tried_${id}`, !isTried);
          render(); // Перерисовываем, чтобы обновить кнопку
        });
      });
    }

    // Слушатели событий
    searchInput.addEventListener('input', render);
    fCountry.addEventListener('change', render);
    fStyle.addEventListener('change', render);
    fRating.addEventListener('change', render);

    // Первый запуск
    render();
  </script>
